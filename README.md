func gcd(int a, int b) -> int {
    while (a && b) {
        if (a > b) {
          a %= b;
        } else {
          b %= a;
        }
    }
    return a + b;
}

func sum (int a, int b) -> int {
    return a + b;
}

func main() -> int  {
    int a = 3, b = 4;
    gcd(a, b);
    sum(2, b);
    do {
        a -= 1;
    } while (a != 1);
}
